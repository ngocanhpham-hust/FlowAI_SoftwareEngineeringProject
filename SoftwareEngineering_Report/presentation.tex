\section{Acceptance Context}
\sectionframe{1. Acceptance Context}{Client need, problem summary, and prototype scope}

% 4
\begin{frame}{Problem Description, Survey, and Scope}
    \small
    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.32\textwidth}
            \fixedblock{Problem Description}{0.45\textheight}{
                \begin{itemize}\tightlist
                    \item Public venues already collect surveillance video.
                    \item Raw footage is mostly reviewed manually.
                    \item Manual review misses repeated crowding patterns and common paths.
                \end{itemize}
            }
        \end{column}
        \begin{column}{0.32\textwidth}
            \fixedblock{Problem Survey}{0.45\textheight}{
                \begin{itemize}\tightlist
                    \item Facility teams need measurable movement data.
                    \item Important outputs include:
                    \begin{itemize}\tightlist
                        \item zone counts and occupancy;
                        \item transition flows;
                        \item heatmaps and alerts.
                    \end{itemize}
                \end{itemize}
            }
        \end{column}
        \begin{column}{0.32\textwidth}
            \fixedblock{Accepted Scope}{0.45\textheight}{
                \begin{itemize}\tightlist
                    \item Local offline-video prototype.
                    \item Upload MP4, configure zones, and run analysis.
                    \item Accepted outputs:
                    \begin{itemize}\tightlist
                        \item dashboard analytics;
                        \item alerts, history, reports.
                    \end{itemize}
                \end{itemize}
            }
        \end{column}
    \end{columns}
    \vspace{0.8em}
    \begin{center}
        \textcolor{ReportBlue}{Privacy constraint: anonymous movement analytics only; no face recognition or identity matching.}
    \end{center}
\end{frame}

    \begin{columns}[T,onlytextwidth]
        \begin{column}{0.48\textwidth}

                    \item Generated processed media and analytics per job.

                    \item Functional and non-functional tests passed in local demo.

            }
        \end{column}
    \end{columns}
\end{frame}
